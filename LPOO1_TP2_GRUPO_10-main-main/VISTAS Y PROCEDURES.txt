    Cli_DNI AS [DNI], 
    Cli_Apellido AS [Apellido], 
    Cli_Nombre AS [Nombre], 
    Cli_Direccion AS [Direccion], 
    OS_CUIT AS [Obrasocial], 
    Cli_NroCarnet AS [NroCarnet]

GO

-- PUNTO 4: Listar ventas realizadas por cliente
IF OBJECT_ID('dbo.listar_ventas_x_cliente_sp', 'P') IS NOT NULL DROP PROCEDURE dbo.listar_ventas_x_cliente_sp;
GO
CREATE PROCEDURE dbo.listar_ventas_x_cliente_sp @cli_dni VARCHAR(20) AS 
BEGIN 
    SET NOCOUNT ON; 
    SELECT 
        V.Ven_Nro AS [Nro_Venta], 
        V.Ven_Fecha AS [Fecha], 
        C.Cli_DNI AS [DNI], 
        C.Cli_Apellido AS [Apellido], 
        C.Cli_Nombre AS [Nombre] 
    FROM Venta V 
    INNER JOIN Cliente C ON C.Cli_DNI = V.Cli_DNI 
    WHERE C.Cli_DNI = @cli_dni 
    ORDER BY V.Ven_Fecha DESC; 
END
GO

-- PUNTO 5: Listar ventas realizadas por rango de fechas
IF OBJECT_ID('dbo.ListarVentasPorFecha', 'P') IS NOT NULL DROP PROCEDURE dbo.ListarVentasPorFecha;
GO
CREATE PROCEDURE dbo.ListarVentasPorFecha
    @fechaDesde DATETIME,
    @fechaHasta DATETIME
AS
BEGIN
    SET NOCOUNT ON;
    SELECT
        V.Ven_Nro,
        V.Ven_Fecha,
        C.Cli_DNI,
        C.Cli_Apellido,
        C.Cli_Nombre
    FROM Venta V
    LEFT JOIN Cliente C ON C.Cli_DNI = V.Cli_DNI
    WHERE V.Ven_Fecha BETWEEN @fechaDesde AND @fechaHasta
    ORDER BY V.Ven_Fecha DESC;
END
GO

-- PUNTO 6: Listar productos vendidos por Cliente
IF OBJECT_ID('dbo.ListarProductosVendidosXCliente', 'P') IS NOT NULL DROP PROCEDURE dbo.ListarProductosVendidosXCliente;
GO
CREATE PROCEDURE dbo.ListarProductosVendidosXCliente @cli_dni VARCHAR(20) AS 
BEGIN 
    SET NOCOUNT ON; 
    SELECT 
        V.Ven_Nro AS [Nro_Venta], 
        V.Ven_Fecha AS [Fecha], 
        P.Prod_Codigo AS [Codigo_Producto], 
        P.Prod_Descripcion AS [Producto], 
        VD.Det_Cantidad AS [Cantidad], 
        VD.Det_Precio AS [Precio_Unitario] 
    FROM Venta V 
    INNER JOIN VentaDetalle VD ON V.Ven_Nro = VD.Ven_Nro 
    INNER JOIN Producto P ON VD.Prod_Codigo = P.Prod_Codigo 
    WHERE V.Cli_DNI = @cli_dni 
    ORDER BY V.Ven_Fecha DESC; 
END
GO

-- PUNTO 7: Listar productos vendidos por rango de fechas
IF OBJECT_ID('dbo.ListarProductosVendidosXFecha', 'P') IS NOT NULL DROP PROCEDURE dbo.ListarProductosVendidosXFecha;
GO
CREATE PROCEDURE dbo.ListarProductosVendidosXFecha @fechaDesde DATETIME, @fechaHasta DATETIME AS 
BEGIN 
    SET NOCOUNT ON; 
    SELECT 
        V.Ven_Nro AS [Nro_Venta], 
        V.Ven_Fecha AS [Fecha], 
        C.Cli_Apellido + ', ' + C.Cli_Nombre AS [Cliente], 
        P.Prod_Descripcion AS [Producto], 
        VD.Det_Cantidad AS [Cantidad], 
        VD.Det_Precio AS [Precio_Unitario] 
    FROM Venta V 
    INNER JOIN Cliente C ON V.Cli_DNI = C.Cli_DNI 
    INNER JOIN VentaDetalle VD ON V.Ven_Nro = VD.Ven_Nro 
    INNER JOIN Producto P ON VD.Prod_Codigo = P.Prod_Codigo 
    WHERE V.Ven_Fecha BETWEEN @fechaDesde AND @fechaHasta 
    ORDER BY V.Ven_Fecha DESC; 
END
GO

-- REQUERIMIENTO EXTRA: Buscar Clientes Avanzado (Buscador)
IF OBJECT_ID('dbo.buscar_clientes', 'P') IS NOT NULL DROP PROCEDURE dbo.buscar_clientes;
GO
CREATE PROCEDURE dbo.buscar_clientes
    @dni      VARCHAR(50),
    @apellido VARCHAR(50),
    @nombre   VARCHAR(50),
    @orden    VARCHAR(20)
AS
BEGIN
    SET NOCOUNT ON;
    -- Concatenamos los comodines '%' para búsquedas parciales
    SET @dni = '%' + @dni + '%';
    SET @apellido = '%' + @apellido + '%';
    SET @nombre = '%' + @nombre + '%';

    IF @orden = 'DNI'
        SELECT * FROM vw_Clientes
        WHERE DNI LIKE @dni 
          AND UPPER(Apellido) LIKE UPPER(@apellido) 
          AND UPPER(Nombre) LIKE UPPER(@nombre)
        ORDER BY DNI ASC;
    ELSE IF @orden = 'Nombre'
        SELECT * FROM vw_Clientes
        WHERE DNI LIKE @dni 
          AND UPPER(Apellido) LIKE UPPER(@apellido) 
          AND UPPER(Nombre) LIKE UPPER(@nombre)
        ORDER BY Nombre ASC;
    ELSE -- Por defecto ordena por Apellido
        SELECT * FROM vw_Clientes
        WHERE DNI LIKE @dni 
          AND UPPER(Apellido) LIKE UPPER(@apellido) 
          AND UPPER(Nombre) LIKE UPPER(@nombre)
        ORDER BY Apellido ASC;
END
GO

    @cuit VARCHAR(20)
AS

        Cli_DNI AS [DNI],
