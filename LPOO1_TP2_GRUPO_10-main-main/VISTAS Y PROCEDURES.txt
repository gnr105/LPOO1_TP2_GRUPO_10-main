-- PUNTO 5: Listar ventas realizadas por rango de fechas

GO
CREATE PROCEDURE dbo.ListarVentasPorFecha
    @fechaDesde DATETIME,
    @fechaHasta DATETIME
AS
BEGIN
    SET NOCOUNT ON;
    SELECT
        V.Ven_Nro,
        V.Ven_Fecha,
        C.Cli_DNI,
        C.Cli_Apellido,
        C.Cli_Nombre
    FROM Venta V
    LEFT JOIN Cliente C ON C.Cli_DNI = V.Cli_DNI
    WHERE V.Ven_Fecha BETWEEN @fechaDesde AND @fechaHasta
    ORDER BY V.Ven_Fecha DESC;
END
GO

-- REQUERIMIENTO EXTRA: Buscar Clientes Avanzado (Buscador)
IF OBJECT_ID('dbo.buscar_clientes', 'P') IS NOT NULL DROP PROCEDURE dbo.buscar_clientes;
GO
CREATE PROCEDURE dbo.buscar_clientes
    @dni      VARCHAR(50),
    @apellido VARCHAR(50),
    @nombre   VARCHAR(50),
    @orden    VARCHAR(20)
AS
BEGIN
    SET NOCOUNT ON;
    -- Concatenamos los comodines '%' para búsquedas parciales
    SET @dni = '%' + @dni + '%';
    SET @apellido = '%' + @apellido + '%';
    SET @nombre = '%' + @nombre + '%';

    IF @orden = 'DNI'
        SELECT * FROM vw_Clientes
        WHERE DNI LIKE @dni 
          AND UPPER(Apellido) LIKE UPPER(@apellido) 
          AND UPPER(Nombre) LIKE UPPER(@nombre)
        ORDER BY DNI ASC;
    ELSE IF @orden = 'Nombre'
        SELECT * FROM vw_Clientes
        WHERE DNI LIKE @dni 
          AND UPPER(Apellido) LIKE UPPER(@apellido) 
          AND UPPER(Nombre) LIKE UPPER(@nombre)
        ORDER BY Nombre ASC;
    ELSE -- Por defecto ordena por Apellido
        SELECT * FROM vw_Clientes
        WHERE DNI LIKE @dni 
          AND UPPER(Apellido) LIKE UPPER(@apellido) 
          AND UPPER(Nombre) LIKE UPPER(@nombre)
        ORDER BY Apellido ASC;
END
GO

-- VENTA: Eliminar venta completa (Detalle + Cabecera en Transacción)
IF OBJECT_ID('dbo.eliminar_venta_sp', 'P') IS NOT NULL DROP PROCEDURE dbo.eliminar_venta_sp;
GO
CREATE PROCEDURE dbo.eliminar_venta_sp
    @nro INT
AS
BEGIN
    SET NOCOUNT ON;
    BEGIN TRANSACTION;
    BEGIN TRY
        DELETE FROM VentaDetalle WHERE Ven_Nro = @nro;
        DELETE FROM Venta WHERE Ven_Nro = @nro;
        COMMIT TRANSACTION;
    END TRY
    BEGIN CATCH
        ROLLBACK TRANSACTION;
        THROW;
    END CATCH
END
GO
